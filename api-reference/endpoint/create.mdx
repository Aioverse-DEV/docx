---
title: 'Create Resource'
api: 'POST https://api.example.com/v1/resources'
description: 'Create a new resource'
---

## Body Parameters

<ParamField body="title" type="string" required>
  The title of the resource
</ParamField>

<ParamField body="content" type="string" required>
  The main content of the resource
</ParamField>

<ParamField body="status" type="string">
  Initial status: `draft` or `published` (default: `draft`)
</ParamField>

<ParamField body="tags" type="array">
  Array of tags to associate with the resource
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata for the resource
  
  <Expandable title="metadata object">
    <ParamField body="category" type="string">
      Resource category
    </ParamField>
    <ParamField body="priority" type="number">
      Priority level (1-5)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the newly created resource
</ResponseField>

<ResponseField name="title" type="string">
  Title of the resource
</ResponseField>

<ResponseField name="content" type="string">
  Main content of the resource
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the resource
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp when the resource was created
</ResponseField>

```python Python
import requests

url = "https://api.example.com/v1/resources"
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}
payload = {
    "title": "New Documentation Page",
    "content": "This is the content of the new page...",
    "status": "draft",
    "tags": ["documentation", "guide"]
}

response = requests.post(url, headers=headers, json=payload)
data = response.json()
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "net/http"
)

func main() {
    payload := map[string]interface{}{
        "title": "New Documentation Page",
        "content": "This is the content of the new page...",
        "status": "draft",
        "tags": []string{"documentation", "guide"},
    }
    
    jsonPayload, _ := json.Marshal(payload)
    req, _ := http.NewRequest("POST", "https://api.example.com/v1/resources", bytes.NewBuffer(jsonPayload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, _ := client.Do(req)
    defer resp.Body.Close()
}
```
